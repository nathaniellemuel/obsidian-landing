init.ts 
// Initialize EOS Web3 client
import { EOSClient, Transaction } from '@eos/sdk';

const client = new EOSClient({
  network: 'mainnet',
  apiKey: process.env.EOS_API_KEY,
});

// Build & send a transaction
async function sendTransfer(
  to: string,
  amount: bigint
): Promise<Transaction> {
  const tx = await client.transaction.create({
    to,
    value: amount,
    gasLimit: BigInt(21000),
  });

  const receipt = await tx.sign().send();
  const confirmed = receipt.status === true;

  if (!confirmed) throw new Error('Transaction failed');
  return receipt;
}
